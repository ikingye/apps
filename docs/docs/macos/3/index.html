    multipass

    canonical/multipass 

    # 创建一个 VM, 1GB 内存和 5GB 磁盘
multipass launch --name myvm --mem 1G --disk 5G

# 为 VM 启动一个 shell
multipass shell myvm

# 挂载
multipass mount ~/code myvm:/home/ubuntu/yewang/code
